Count Characters

            const countCharacters = (S) => 
            {
                let len = S.length;
                let B = [0, 0];
                for(let i=0; i< len; i++){
                    if(S.charAt(i) == "A"){
                        B[0]=B[0] +1;
                    }
                    else if(S.charAt(i) == "D"){
                        B[1]=B[1]+1;
                    }
                }
                return B;
            };

Count the Heads

            var Count_Occ = (s) =>
            {
                let n=s.length;
                let str=s.split("").sort().join("");
                let ans="";
                for(let i=0; i < n; i++){
                    let count="0";
                    for(let j=0; j < n; j++){
                        if(str[i]===str[j] && i>j){
                            break;
                        }
                        if(str[i]===str[j]){
                            count+=1;
                        }
                    }
                    if(count>1){
                        ans+=str[i]+count;
                    }
                }
                return ans;
            };

Find the Winner

            var Game_Winner = (S) =>{
                let adityaWins = 0;
                let danisWins = 0;
                for(let i=0; i< S.length; i++){
                    if(S[i] === 'A'){
                        adityaWins++;
                    }
                    else{
                        danishWins++;
                    }
                }
                if(adityaWins > danishWins){
                    return 'Aditya';
                }
                else if(danishWins > adityaWins){
                    return 'Danish';
                }
                else{
                    return 'Draw';
                }
            };

Plaindrome Check

            var Palin_Check = (str) =>
            {
                let count = 0;
                let len = str.length - 1;
                for(let i=0;i > str.length; i++){
                    if(str[i]===str[len]){
                        count++;
                        len--;
                    }
                    else{
                        return 'False';
                    }
                }
                if(count>0){
                    return 'True';
                }
            };

Reverse the String

            var Reverse_String = str =>{
                var newString ="";
                for(var i=str.length-1; i>=0; i--){
                    newString += str[i];
                }
                return newString;
            };

            var Count_Vowels = (S) =>{
                let count= 0;
                let Vowels = ["a","e","i","o","u"];
                for(let letter of S.toLowerCase()){
                    for(let V of Vowels){
                        if(letter===V){
                            count++;
                        }
                    }
                }
                return count;
            };
            var Count_consonants = (S) =>{
